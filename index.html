border: none; color: #8a8b8c; font-size: 12px; font-weight: normal; padding: 0;") forgot_btn.setCursor(Qt.CursorShape.PointingHandCursor) forgot_btn.clicked.connect(self.on_forgot_password_clicked) links_layout.addWidget(forgot_btn, alignment=Qt.AlignmentFlag.AlignLeft) reg_link_btn = QPushButton("Нет аккаунта? Зарегистрируйте!") reg_link_btn.setStyleSheet("background: none; border: none; color: #f0eedb; font-size: 12px; font-weight: normal; padding: 0;") reg_link_btn.setCursor(Qt.CursorShape.PointingHandCursor) reg_link_btn.clicked.connect(lambda: self.change_page_smooth(2)) links_layout.addWidget(reg_link_btn, alignment=Qt.AlignmentFlag.AlignRight) layout.addLayout(links_layout) layout.addSpacing(15) self.login_button = QPushButton("ВОЙТИ") self.login_button.clicked.connect(self.on_login_clicked) layout.addWidget(self.login_button) page.setLayout(layout) self.pages.addWidget(page) def _setup_menu_page(self): """Экран 2: Внутреннее меню лаунчера после успешного входа""" page = QWidget() layout = QVBoxLayout() layout.setSpacing(15) layout.setContentsMargins(35, 45, 35, 45) game_title = QLabel("HEART OF WAR") game_title.setFont(QFont("Arial", 24, QFont.Weight.Bold)) game_title.setAlignment(Qt.AlignmentFlag.AlignCenter) layout.addWidget(game_title) version_label = QLabel("v. 1.0.0") version_label.setFont(QFont("Arial", 10)) version_label.setStyleSheet("color: #8a8b8c;") version_label.setAlignment(Qt.AlignmentFlag.AlignCenter) layout.addWidget(version_label) layout.addSpacing(20) self.welcome_label = QLabel("Добро пожаловать, Боец!") self.welcome_label.setFont(QFont("Arial", 13)) self.welcome_label.setAlignment(Qt.AlignmentFlag.AlignCenter) self.welcome_label.setStyleSheet("color: #f0eedb;") layout.addWidget(self.welcome_label) layout.addSpacing(30) self.start_game_button = QPushButton("ЗАПУСТИТЬ ИГРУ") self.start_game_button.clicked.connect(self.on_start_game_clicked) layout.addWidget(self.start_game_button) self.logout_button = QPushButton("СМЕНИТЬ АККАУНТ") self.logout_button.setStyleSheet("background-color: #2d2e30; color: #ffffff; border: 1px solid #4a4b4c;") self.logout_button.clicked.connect(lambda: self.change_page_smooth(0)) layout.addWidget(self.logout_button) page.setLayout(layout) self.pages.addWidget(page) def _setup_register_page(self): """Экран 3: Окно регистрации с загрузкой PNG-иконок и эффектами затемнения""" page = QWidget() layout = QVBoxLayout() layout.setSpacing(12) layout.setContentsMargins(35, 40, 35, 40) title_label = QLabel("STEEL CLUB ID") title_label.setFont(QFont("Arial", 26, QFont.Weight.Bold)) title_label.setAlignment(Qt.AlignmentFlag.AlignCenter) layout.addWidget(title_label) reg_label = QLabel("РЕГИСТРАЦИЯ И СВЯЗЬ") reg_label.setFont(QFont("Arial", 11, QFont.Weight.Bold)) reg_label.setStyleSheet("color: #8a8b8c; letter-spacing: 2px;") reg_label.setAlignment(Qt.AlignmentFlag.AlignCenter) layout.addWidget(reg_label) layout.addSpacing(15) base_dir = os.path.dirname(__file__) google_icon_path = os.path.join(base_dir, "assets", "icons", "google.png") discord_icon_path = os.path.join(base_dir, "assets", "icons", "discord.png") vk_icon_path = os.path.join(base_dir, "assets", "icons", "vk.png") # Кнопка быстрого входа через GOOGLE google_btn = QPushButton("Войти через Google") google_btn.setFont(QFont("Arial", 12)) if os.path.exists(google_icon_path): google_btn.setIcon(QIcon(google_icon_path)) google_btn.setIconSize(QSize(24, 24)) google_btn.setStyleSheet(""" QPushButton { background-color: #ffffff; color: #1a1b1c; border: 1px solid #dcdcdc; padding: 12px; text-align: left; } QPushButton:hover { background-color: #e5e5e5; color: #1a1b1c; } QPushButton:pressed { background-color: #dcdcdc; } """) google_btn.clicked.connect(lambda: self.on_social_login_clicked("Google")) layout.addWidget(google_btn) # Кнопка быстрого входа через ВКОНТАКТЕ (С белой иконкой и затемнением) vk_btn = QPushButton("Войти через ВКонтакте") vk_btn.setFont(QFont("Arial", 12)) if os.path.exists(vk_icon_path): pixmap_vk = QPixmap(vk_icon_path) painter_vk = QPainter(pixmap_vk) painter_vk.setCompositionMode(QPainter.CompositionMode.CompositionMode_SourceIn) painter_vk.fillRect(pixmap_vk.rect(), QColor(255, 255, 255)) painter_vk.end() vk_btn.setIcon(QIcon(pixmap_vk)) vk_btn.setIconSize(QSize(24, 24)) vk_btn.setStyleSheet(""" QPushButton { background-color: #0077FF; color: #ffffff; border: 1px solid #0066DD; padding: 12px; text-align: left; } QPushButton:hover { background-color: #0066DD; color: #ffffff; } QPushButton:pressed { background-color: #0055CC; } """) vk_btn.clicked.connect(lambda: self.on_social_login_clicked("ВКонтакте")) layout.addWidget(vk_btn) # Кнопка быстрого входа через DISCORD (С белой иконкой) discord_btn = QPushButton("Войти через Discord") discord_btn.setFont(QFont("Arial", 12)) if os.path.exists(discord_icon_path): pixmap_ds = QPixmap(discord_icon_path) painter_ds = QPainter(pixmap_ds) painter_ds.setCompositionMode(QPainter.CompositionMode.CompositionMode_SourceIn) painter_ds.fillRect(pixmap_ds.rect(), QColor(255, 255, 255)) painter_ds.end() discord_btn.setIcon(QIcon(pixmap_ds)) discord_btn.setIconSize(QSize(24, 24)) discord_btn.setStyleSheet(""" QPushButton { background-color: #5865F2; color: #ffffff; border: 1px solid #404eed; padding: 12px; text-align: left; } QPushButton:hover { background-color: #4752c4; } QPushButton:pressed { background-color: #3c45a5; } """) discord_btn.clicked.connect(lambda: self.on_social_login_clicked("Discord")) layout.addWidget(discord_btn) layout.addSpacing(25) back_to_login_btn = QPushButton("УЖЕ ЕСТЬ АККАУНТ? ВОЙТИ") back_to_login_btn.setStyleSheet("background-color: #2d2e30; color: #ffffff; border: 1px solid #4a4b4c;") back_to_login_btn.clicked.connect(lambda: self.change_page_smooth(0)) layout.addWidget(back_to_login_btn) page.setLayout(layout) self.pages.addWidget(page) def on_login_clicked(self): username = self.login_input.text().strip() password = self.password_input.text().strip() if not username or not password: QMessageBox.warning(self, "Ошибка ввода", "Пожалуйста, заполните логин и пароль!") return self.save_credentials(username, password) self.welcome_label.setText(f"Приветствуем в Штабе, {username}!") self.session_token = f"token_for_{username}_secure" self.change_page_smooth(1) def on_start_game_clicked(self): game_path = GAME_EXE_PATH game_dir = os.path.dirname(game_path) if not os.path.exists(game_path): QMessageBox.critical(self, "Ошибка запуска", f"Игра '{GAME_NAME}' не обнаружена по пути:\n{game_path}\nПроверьте подключение диска G или имя файла!") return try: creation_flags = 0 if sys.platform == "win32": creation_flags = subprocess.CREATE_NO_WINDOW subprocess.Popen([game_path, f"--token={self.session_token}"], cwd=game_dir, creationflags=creation_flags) self.close() sys.exit() except Exception as e: QMessageBox.critical(self, "Критическая ошибка", f"Не удалось запустить игровой движок:\n{str(e)}") def on_forgot_password_clicked(self): QMessageBox.information(self, "Восстановление", "Функция восстановления пароля будет доступна после запуска официального веб-сайта Steel Club ID.") def on_social_login_clicked(self, platform_name): if platform_name == "Google": # Перенаправляем игрока на твой собственный защищенный сайт на Vercel webbrowser.open(GAME_WEBSITE_URL) else: QMessageBox.information(self, "Вход через соцсети", f"Вход через {platform_name} будет настроен сразу после привязки базы данных к сайту игры.") def save_credentials(self, username, password): try: data = {"username": username, "password": password} with open(CONFIG_FILE, "w", encoding="utf-8") as f: json.dump(data, f, ensure_ascii=False, indent=4) except Exception as e: pass def load_saved_credentials(self): if os.path.exists(CONFIG_FILE): try: with open(CONFIG_FILE, "r", encoding="utf-8") as f: data = json.load(f) if "username" in data: self.login_input.setText(data["username"]) if "password" in data: self.password_input.setText(data["password"]) except Exception as e: pass if __name__ == "__main__": app = QApplication(sys.argv) launcher = SteelClubLauncher() launcher.show() sys.exit(app.exec())
